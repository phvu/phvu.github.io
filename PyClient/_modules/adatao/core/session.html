    """
    Open a Session to a server with given credential
    :param host: IP address or host name of pAnalytics
    :param port: Server port, default to 16000
    :param username: username
    :param password: password

[docs]    def login(self, username='anonymous', password=''):
        """
        Login to the server using given username and password
        :param username:
        :param password:
        :return: NA
        :exception: raise {@code PyClientServerException} if username/password doesn't match
        Raise {@code ConnectionError} if there is problem connecting to the server.
        """
        # This is not very Pythonic (we're all grown-ups), but well...
        util.check_type(username, 'username', str, unicode)
        util.check_type(password, 'password', str, unicode)

        ret = self.cli.exec_command('connect', {'credentials': '%s:%s' % (password, username)})
        self.cli.token = ret['accessToken']

[docs]    def list_ddfs(self):
        """
        Show current list of DDFs
        """
        res = self.cli.exec_command('ListDDF', {'test': 'test'})['result']
        df_ret = pd.DataFrame(columns=['owner', 'name', 'created', 'source'])

        safe_get = lambda dic, k, default_val: default_val if k not in dic else dic[k]

        for ddf_info in res:
            df_ret = df_ret.append({'name':     safe_get(ddf_info, 'name', ''),
                                    'created':  ddf_info['createdTime'],
                                    'source':   safe_get(ddf_info, 'dataSourceProfile', None),
                                    'owner':    safe_get(ddf_info, 'owner', None)},
                                   ignore_index=True)
        return df_ret

[docs]    def sql(self, sql_str):
        """
        Run a SQL command on server
        :param sql_str: The SQL command to be run
        :return: a pandas DataFrame if given a SELECT query, otherwise a string.
        """
        # TODO: revise this
        util.check_type(sql_str, 'sql_str', str, unicode)

        # only take the first command
        sql_str = sql_str.split(';')[0]
        sql_str_lower = sql_str.lower()

        if sql_str_lower.startswith('select'):
            res = self.sql2ddf(sql_str)
            return res.head(res.rows)
        else:
            res = self.cli.exec_command('Sql2ListString', {'sqlCmd': sql_str})['result']['results']
            if sql_str_lower.startswith('describe'):
                res = [r.split('\t')[0:2] for r in res]
            return res



[docs]    def sql2ddf(self, sql_str, cached=True):
        """
        Run a select query and return a DistributedDataFrame
        :param sql_str: the SQL query to be run
        :param cached:
        :return: A DistributedDataFrame object if sql_str is a SELECT query, else raise a ValueError.
        """
        util.check_type(sql_str, 'sql_str', str, unicode)
        util.check_type(cached, 'cached', bool)

        # only take the first command
        sql_str = sql_str.split(';')[0]

        if sql_str.lower().startswith('select'):
            return self.load('sql', sqlCmd=sql_str, cache=cached)
        else:
            raise ValueError('Only SELECT queries are allowed!')

[docs]    def load_from_hdfs(self, path, schema, serde, original_source='hdfs', is_json=False):
        """
        Load data from HDFS into a DistributedDataFrame
        :param path: HDFS path
        :param schema: data schema
        :param serde: The Serializer/Deserializer name
        :param original_source: either `hdfs` or `local`. The `local` value is temporarily
        here for Narratives data uploading usage to specify that the data is really from users' local filesystems
        and is uploaded by Narratives to HDFS.
        :return: a DistributedDataFrame object
        """
        if (not isinstance(original_source, str)) or original_source.lower() not in ['hdfs', 'local']:
            raise ValueError('Invalid source. Only hdfs or local is supported')

        return self.load('hdfs', uri=path, schema=schema, serde=serde,
                         originalSource=original_source, isJSON=is_json)

        """

        :param uri: S3 path

        :param aws_secret_key: AWS Secret Key
        :param schema: data schema
        :param serde: The Serializer/Deserializer name
        :return: a DistributedDataFrame object
        """
        if (not is_json) and schema is None:
            raise ValueError('A valid schema is needed for the dataset')
        return self.load('S3', uri=uri, awsKeyID=aws_key_id, awsSecretKey=aws_secret_key,
                         schema=schema, serde=serde, isJSON=is_json)
