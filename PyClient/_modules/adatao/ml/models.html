[docs]    def from_json(json_data):
        """
        parse model from com.adatao.ddf.ml.Model's json
        :return:
        """
        model_name = json_data['mName']
        model_type = json_data['modelType']
        raw_model = json_data['mRawModel']
        columns = json_data['mTrainedColumns']

        if model_type == "com.adatao.ML.spark.clustering.KMeansModel":
            return KMeansModel(model_name, columns, raw_model)
        else:
            raise NotImplementedError('Unrecognized model: ' + model_type)

[docs]    def predict(self, data, cols=None):
        colnames = data.colnames
        col_idx = util.parse_columns(colnames, self.cols) if cols is None else util.parse_columns(colnames, cols)

        res = data.client.exec_command('XsYpred',
                                       {'dataContainerID': data.oid,
                                        'xCols': col_idx,
                                        'modelID': self.model_id})['result']
        return create_distributed_data_frame(data.client, res)

[docs]class KMeansModel(ClusteringModel):

    def __init__(self, model_name, columns, raw_model):
        super(KMeansModel, self).__init__(model_name, columns)

        centers = pd.DataFrame(columns=columns)
        for i in range(0, len(raw_model['clusterCenters'])):
            centers = centers.append(dict(zip(columns, raw_model['clusterCenters'][i]['values'])), ignore_index=True)

        self._centers = centers
        self.wcss = raw_model['wcss']

    @property
    def centers(self):
        return self._centers.copy()

    def __repr__(self):
        s = 'Kmeans model with centers:\n' + self._centers.to_string()
        return s
