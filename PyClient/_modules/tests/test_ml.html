[docs]    def testClustering(self):
        kmeans_model = ml.clustering.kmeans(self.mtcars, ['cyl', 'mpg'], num_clusters=2,
                                            max_iterations=10, distance='cosine')
        ddf2 = self.mtcars[['cyl', 'mpg']].result
        ddf3 = kmeans_model.predict(ddf2)
        self.assertEqual(ddf3.cols, 3)
        self.assertEqual(ddf3.colnames[-1], 'yPredict')

        ddf4 = ml.clustering.silhouette(ddf3, 'yPredict', 'cosine')
        self.assertEqual(ddf4.cols, 1)
        self.assertEqual(ddf4.colnames[0], 'silhouette')
        avg_silhouette = ddf4.mean(1)
        self.assertAlmostEqual(0.752792, avg_silhouette, 5)
