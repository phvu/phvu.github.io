[docs]    def testSql(self):
        ls_tables = self.session.sql('show tables')
        self.assertIsInstance(ls_tables, list)

        if len(ls_tables) > 0:
            tb_name = ls_tables[0]
            obj = self.session.sql('describe %s' % tb_name)
            self.assertIsInstance(obj, list)

            obj = self.session.sql('select count(*) from %s' % tb_name)
            self.assertIsInstance(obj, pd.DataFrame)
            n = int(obj.iloc[0, 0])
            if n > 0:
                obj = self.session.sql('select * from %s limit 1' % tb_name)
                self.assertIsInstance(obj, pd.DataFrame)
                self.assertEqual(len(obj), 1)

            for tb in ls_tables:
                obj = self.session.sql2ddf('select * from %s limit 1' % tb)
                self.assertIsInstance(obj, DistributedDataFrame)
                if obj.rows == 1:
                    break
        else:
            with self.assertRaises(PyClientServerException):
                self.session.sql('describe bullshit')

        with self.assertRaises(ValueError):
            self.session.sql2ddf('show tables')
