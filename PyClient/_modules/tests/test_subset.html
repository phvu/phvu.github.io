[docs]    def testSubset(self):

        n_rows = self.mtcars.rows
        columns = self.mtcars.colnames[:]

        df1_exp = self.mtcars[:]
        self.assertIsInstance(df1_exp, adatao.SubsetExpression)
        df1 = df1_exp.result
        self.assertIsInstance(df1, adatao.DistributedDataFrame)
        self.assertEqual(df1.rows, n_rows)
        self.assertEqual(df1.cols, self.mtcars.cols)
        self.assertItemsEqual(columns, df1.colnames)

        df2 = self.mtcars[:4].result
        self.assertEqual(df2.rows, n_rows)
        self.assertEqual(df2.cols, 4)
        self.assertItemsEqual(columns[:4], df2.colnames)

        df3 = self.mtcars[:, :].result
        self.assertEqual(df3.rows, n_rows)
        self.assertEqual(df3.cols, self.mtcars.cols)
        self.assertItemsEqual(columns, df3.colnames)

        df4 = self.mtcars[:, 'mpg'].result
        self.assertEqual(df4.rows, n_rows)
        self.assertEqual(df4.cols, 1)
        self.assertItemsEqual(['mpg'], df4.colnames)

        df5 = self.mtcars[:, ['mpg', 'cyl']].result
        self.assertEqual(df5.rows, n_rows)
        self.assertEqual(df5.cols, 2)
        self.assertItemsEqual(['mpg', 'cyl'], df5.colnames)

        with self.assertRaises(KeyError):
            # any custom slicing on the filter expression (rows) is not supported
            df6 = self.mtcars[1:4, 'mpg'].result

        # "Natural" column access using pandas-like syntax
        df7_exp = self.mtcars.mpg
        self.assertIsInstance(df7_exp, adatao.SubsetExpression)
        df7 = df7_exp.result
        self.assertIsInstance(df7, adatao.DistributedDataFrame)
        self.assertEqual(df7.rows, n_rows)
        self.assertEqual(df7.cols, 1)
        self.assertItemsEqual(['mpg'], df7.colnames)

        # Less "natural" column access using pandas-like syntax
        df8 = self.mtcars['mpg'].result
        self.assertEqual(df8.rows, n_rows)
        self.assertEqual(df8.cols, 1)
        self.assertItemsEqual(['mpg'], df8.colnames)

        # Access the columns using a list of column names, similar to [:, ['mpg', 'disp']]
        df9 = self.mtcars[['mpg', 'disp']].result
        self.assertEqual(df9.rows, n_rows)
        self.assertEqual(df9.cols, 2)
        self.assertItemsEqual(['mpg', 'disp'], df9.colnames)

        # Access the columns using a list of column indices, similar to [:, [2, 3, 5]]
        df10 = self.mtcars[[2, 3, 5]].result
        self.assertEqual(df10.rows, n_rows)
        self.assertEqual(df10.cols, 3)
        self.assertItemsEqual([columns[x] for x in [2, 3, 5]], df10.colnames)

        # "complicated" filter expression and columns
        df11 = self.mtcars[self.mtcars.mpg > 3, [2, 5]].result
        self.assertEqual(df11.rows, n_rows)
        self.assertEqual(df11.cols, 2)
        self.assertItemsEqual([columns[x] for x in [2, 5]], df11.colnames)

        df12 = self.mtcars[(self.mtcars.mpg > 3) & (self.mtcars['disp'] < 0), ['mpg', 'disp', 'cyl']].result
        self.assertEqual(df12.rows, 0)
        self.assertEqual(df12.cols, 3)
        self.assertItemsEqual(['mpg', 'disp', 'cyl'], df12.colnames)

        df13 = self.mtcars[(self.mtcars.mpg > 3) | (self.mtcars['disp'] < 0.0), ['mpg', 'disp', 'cyl']].result
        self.assertEqual(df13.rows, n_rows)
        self.assertEqual(df13.cols, 3)
        self.assertItemsEqual(['mpg', 'disp', 'cyl'], df13.colnames)

        with self.assertRaises(ValueError):
            df14 = self.mtcars[self.mtcars[:, ['mpg', 'disp']] < 0, ['mpg', 'disp', 'cyl']].result

[docs]    def testSubsetStr(self):

        n_rows = self.mtcars.rows
        columns = self.mtcars.colnames[:]

        df12 = self.mtcars['(mpg > 3) & (disp < 0)', ['mpg', 'disp', 'cyl']].result
        self.assertEqual(df12.rows, 0)
        self.assertEqual(df12.cols, 3)
        self.assertItemsEqual(['mpg', 'disp', 'cyl'], df12.colnames)

        df13 = self.mtcars['(mpg > 3) | (disp < 0.0)', ['mpg', 'disp', 'cyl']].result
        self.assertEqual(df13.rows, n_rows)
        self.assertEqual(df13.cols, 3)
        self.assertItemsEqual(['mpg', 'disp', 'cyl'], df13.colnames)

        with self.assertRaises(adatao.PyClientParserException):
            df14 = self.mtcars['mpg_stupid_column_that_sdf < 0', ['mpg', 'disp', 'cyl']].result

        df15 = self.mtcars['(~mpg.isnull()) & (cyl > 5)', ['mpg', 'cyl']].result
        # self.assertEqual(df15.rows, 21)
        self.assertEqual(df15.cols, 2)
        self.assertItemsEqual(['mpg', 'cyl'], df15.colnames)

        with self.assertRaises(adatao.PyClientParserException):
            # "and" and "or" DON'T work. Use bitwise operators (&, |) instead
            df16 = self.mtcars['(~mpg.isnull()) and (cyl > 5)', ['mpg', 'cyl']].result

        df18 = self.mtcars['qesc.contains("16.", ignore_case=False)', :].result
        # df19 = self.mtcars['qesc.contains("16.", ignore_case=True)', :].result

        with self.assertRaises(adatao.PyClientParserException):
            df20 = self.mtcars['mpg and qesc.contains("16.", ignore_case=True)', :].result
