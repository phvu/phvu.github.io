[docs]    def testTransform(self):

        # transform on an immutable DDF
        self.mtcars.is_mutable = False
        df1 = self.mtcars.transform('Hive', 'cyl2=cyl*2', 'hp2 = hp/5')
        self.assertEqual(self.mtcars.cols, 11)
        self.assertEqual(df1.cols, 13)
        self.assertEqual(self.mtcars.rows, df1.rows)
        d_head = df1.head()
        for i in d_head.index:
            self.assertAlmostEqual(d_head.loc[i, 'cyl2'], 2*d_head.loc[i, 'cyl'])
            self.assertAlmostEqual(d_head.loc[i, 'hp2'], d_head.loc[i, 'hp']/5)

        # reset this
        self.mtcars.is_mutable = True

        # transform on a mutable DDF
        self.assertTrue(df1.is_mutable)
        df2 = df1.transform('Hive', 'carb2=carb*2')
        self.assertEqual(df2.rows, df1.rows)
        self.assertEqual(df2.cols, 14)
        self.assertEqual(df1.oid, df2.oid)
        self.assertItemsEqual(df1.colnames, df2.colnames)

        d1 = self.mtcars.head(13)
        self.assertIsInstance(d1, pd.DataFrame)
        self.assertEqual(len(d1), 13)
        self.assertEqual(len(d1.columns), 11)

        d2 = self.mtcars.sample(2)
        self.assertIsInstance(d2, pd.DataFrame)
        self.assertEqual(len(d2), 2)
        self.assertEqual(len(d2.columns), 11)

        df1 = self.mtcars.sample2ddf(0.2)
        self.assertIsInstance(df1, DistributedDataFrame)
        self.assertEqual(df1.cols, self.mtcars.cols)
        self.assertNotEqual(self.mtcars.oid, df1.oid)

        df2 = self.mtcars.sample2ddf(0.2, replace=True)
        self.assertIsInstance(df2, DistributedDataFrame)
        self.assertEqual(df2.cols, self.mtcars.cols)
        self.assertNotEqual(self.mtcars.oid, df2.oid)

        with self.assertRaises(PyClientServerException):
            # bigger than 1
            self.mtcars.sample2ddf(100)

[docs]    def testSort(self):
        df1 = self.mtcars.sort('mpg', descending=False)
        self.assertIsInstance(df1, DistributedDataFrame)
        self.assertEqual(df1.rows, self.mtcars.rows)
        self.assertEqual(df1.cols, self.mtcars.cols)
        d1 = df1.head()
        for i in range(0, len(d1)-1):
            assert d1.iloc[i, 0] <= d1.iloc[i+1, 0]

        df2 = self.mtcars.sort('mpg', descending=True)
        self.assertIsInstance(df2, DistributedDataFrame)
        self.assertEqual(df2.rows, self.mtcars.rows)
        self.assertEqual(df2.cols, self.mtcars.cols)
        d2 = df2.head()
        for i in range(0, len(d2)-1):
            assert d2.iloc[i, 0] >= d2.iloc[i+1, 0]

        with self.assertRaises(ValueError):
            self.mtcars.sort('stupid column which will raise ValueError')

[docs]    def testScale(self):
        df1 = self.mtcars.scale('rescaling')
        self.assertEqual(df1.rows, self.mtcars.rows)
        self.assertEqual(df1.cols, self.mtcars.cols)
        d1 = df1.head(df1.rows)
        for c in d1.columns:
            self.assertAlmostEqual(d1[c].min(), 0.0)
            self.assertAlmostEqual(d1[c].max(), 1.0)

        df2 = self.mtcars.scale('normalization')
        self.assertEqual(df2.rows, self.mtcars.rows)
        self.assertEqual(df2.cols, self.mtcars.cols)
        d2 = df2.head(df2.rows)
        for c in d2.columns:
            self.assertAlmostEqual(d2[c].mean(), 0.0)
            self.assertAlmostEqual(d2[c].std(), 1.0)

        with self.assertRaises(ValueError):
            self.mtcars.scale('Normalization')
