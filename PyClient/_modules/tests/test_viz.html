        viz.set_renderer_mode('html', output_dir=self.out_dir)
        for s in glob.glob(self.out_dir + '/*.html'):
            os.remove(s)

[docs]    def testViz(self):
        mtcars = self.df_mtcars
        viz.bar_chart(data=mtcars, category='cyl', measurement='mpg')

        # barChart with titles
        viz.bar_chart(data=mtcars, category='cyl', measurement='mpg',
                      title="Relationship between mpg and cyl",
                      columnXTitle="cylinder", columnYTitle="miles per gallon")

        # grouped barChart
        viz.bar_chart(data=mtcars, category='cyl', measurement='mpg', category2='gear', grouped=True)

        # stacked barChart
        viz.bar_chart(data=mtcars, category='cyl', measurement='mpg', category2='gear', stacked=True)

        # multiple barCharts
        viz.multiple_bar_chart(data=mtcars, category='cyl', measurement='mpg', category2='gear')

        # lineChart
        viz.line_chart(data=mtcars, category='cyl', measurement='mpg')

        # stacked lineChart
        viz.line_chart(data=mtcars, category='cyl', measurement='mpg', category2='gear', stacked=True)

        # areaChart
        viz.area_chart(data=mtcars, category='cyl', measurement='mpg')

        # stacked areaChart
        viz.area_chart(data=mtcars, category='cyl', measurement='mpg', category2='gear', stacked=True)

        # boxPlot
        viz.box_plot(mtcars, category='cyl', measurement='mpg')

        # pieChart
        data = mtcars.reset_index().groupby('cyl').count()['mpg'].reset_index()
        viz.pie_chart(data=data, category='cyl', measurement='mpg')

        # donutChart
        viz.pie_chart(data=data, category='cyl', measurement='mpg', donut=True)

        # heatMap
        data = mtcars.groupby(['cyl', 'gear']).count()['mpg'].reset_index()
        viz.heat_map(data, category='cyl', measurement='gear', category2='mpg')

        # scatterChart
        viz.scatterplot(mtcars, category='mpg', measurement='disp')
        viz.scatterplot(mtcars, category='mpg', measurement='disp', category2='cyl')

        # treeMap
        viz.tree_map(data=mtcars, category="cyl", measurement="mpg")
        # grouped treeMap
        viz.tree_map(data=mtcars, category="cyl", measurement="mpg", category2="gear")

        # data table
        viz.data_table(data=mtcars)

        # autoChart
        # dataTable
        viz.auto_chart(mtcars)

        # scatterChart
        viz.auto_chart(mtcars[["mpg", "cyl"]])

        data = mtcars.copy()
        data.gear = data.gear.astype(str)
        data.cyl = data.cyl.astype(str)
        data.mpg = data.mpg.astype(float)

        # barChart
        viz.auto_chart(data[["mpg", "gear"]])

        # grouped barChart
        viz.auto_chart(data[["mpg", "cyl", "gear"]])
