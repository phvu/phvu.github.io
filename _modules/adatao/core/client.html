        self.host = host

        self.session = requests.Session()

[docs]    def exec_command(self, command, params):
        if command != 'connect' and self.token == '':
            raise PyClientException('Please login using adatao.connect() or Session.login()')

        params_json = json.dumps(params)

        # logging.log(logging.DEBUG, "Executing " + command + "(" + params_json + ")")

        r = self.session.post('http://%s:%s/command/%s' % (self.host, str(self.port), command),
                              params_json, headers={"X-Access-Token": self.token})

        if r.status_code != 200:
            raise PyClientServerException('Get failure status code from server: Response HTTP code=%d'
                                          % r.status_code,
                                          r.status_code, command, params_json)

        try:
            r = r.json()
            res = json.loads(r['result'])
        except ValueError:
            if r.text.endswith('\nnull\n'):
                # This is due to a bug in server code
                res = json.loads(json.loads(r.text[:-6])['result'])
            else:
                print 'Couldn\'t decode server response:'
                print r.text
                raise

        if 'success' in res and res['success']:
            return res
        else:
            if 'message' in res and 'code' in res:
                raise PyClientServerException(res['message'], res['code'], command, params_json)
            else:
                raise PyClientServerException(str(res), -1, command, params_json)
