[docs]class PyClientServerException(PyClientException):
    """
    Represents error when executing commands on server side
    """
    def __init__(self, server_message, server_code, command=None, params=None):
        self.message = server_message
        self.code = server_code
        self.command = command
        self.params = params

    def __str__(self):
        return '%s (code = %s, command = %s)' % (self.message, str(self.code), str(self.command))

    def __unicode__(self):
        return u'%s (code = %s, command = %s)' % (self.message, unicode(self.code), unicode(self.command))
