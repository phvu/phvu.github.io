    @staticmethod
    def _parse_contains_args(args, kws):
        if len(args) == 1:
            if isinstance(args[0], ast.Str):
                if len(kws) == 0:
                    return args[0].s, None
                elif len(kws) == 1:
                    if kws[0].arg == 'ignore_case' and isinstance(kws[0].value, ast.Name) \
                            and kws[0].value.id in ['True', 'False']:
                        return args[0].s, kws[0].value.id == 'True'
        elif len(args) == 2:
            if isinstance(args[0], ast.Str) and \
               isinstance(args[1], ast.Name) and args[1].id in ['True', 'False']:
                return args[0].s, args[1].id == 'True'

        raise PyClientParserException('Invalid arguments for contains()')


[docs]    def visit_Call(self, node):
        if not isinstance(node.func, ast.Attribute):
            raise PyClientParserException('Only support a function call on a column. Offset = %d'
                                          % node.func.col_offset)
        col_name = node.func.value.id
        func_name = node.func.attr
        if col_name not in self.ddf.colnames:
            raise PyClientParserException('Expect a column name: %s' % col_name)
        if func_name == 'isnull':
            self.expressions.append(self.ddf[col_name].isnull())
        elif func_name == 'contains':
            params = FilterExpressionVisitor._parse_contains_args(node.args, node.keywords)
            if params[1] is None:
                self.expressions.append(self.ddf[col_name].contains(params[0]))
            else:
                self.expressions.append(self.ddf[col_name].contains(params[0], ignore_case=params[1]))
        else:
            raise PyClientParserException('Unknown function: %s' % func_name)



[docs]    def visit_Compare(self, node):
        if len(node.ops) != 1 or len(node.comparators) != 1:
            raise PyClientParserException('Only support comparison between a column and a value. Operator = %s'
                                          % node.ops[0])

        self.visit(node.left)
        self.visit(node.comparators[0])

        operand_right = self.expressions.pop()
        operand_left = self.expressions.pop()
        if not isinstance(operand_left, SubsetExpression):
            raise PyClientParserException('Only support comparison between a column and a value. Operator = %s'
                                          % node.ops[0])
        op_type = type(node.ops[0])

        if op_type is ast.Gt:
            subset_exp = operand_left > operand_right
        elif op_type is ast.GtE:
            subset_exp = operand_left >= operand_right
        elif op_type is ast.Lt:
            subset_exp = operand_left < operand_right
        elif op_type is ast.LtE:
            subset_exp = operand_left <= operand_right
        elif op_type is ast.Eq:
            subset_exp = operand_left == operand_right
        elif op_type is ast.NotEq:
            subset_exp = operand_left != operand_right
        else:
            raise PyClientParserException('Unsupported comparison operator: %s' % op_type)
        self.expressions.append(subset_exp)



[docs]    def visit_UnaryOp(self, node):
        self.visit(node.operand)
        op_type = type(node.op)
        operand = self.expressions.pop()

        if op_type is ast.Invert:
            if not isinstance(operand, SubsetExpression):
                raise PyClientParserException('Only support invert (~) on a subset expression')
            self.expressions.append(~operand)
        else:
            raise ValueError('Unsupported unary operator: %s' % op_type)



[docs]    def visit_BinOp(self, node):
        self.visit(node.left)
        self.visit(node.right)

        op_right = self.expressions.pop()
        op_left = self.expressions.pop()
        op_type = type(node.op)

        if (not isinstance(op_left, SubsetExpression)) or (not isinstance(op_right, SubsetExpression)):
            raise PyClientParserException('Only support binary operators between 2 subset expression. Operator = %s' %
                                          op_type)
        if op_type is ast.BitAnd:
            self.expressions.append(op_left & op_right)
        elif op_type is ast.BitOr:
            self.expressions.append(op_left | op_right)
        else:
            raise PyClientParserException('Unsupported binary operator: %s' % type(node.op))
