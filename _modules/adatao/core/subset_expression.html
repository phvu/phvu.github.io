    index = -1
    name = ''

    def __init__(self, index=-1, name=''):
        self.index = index
        self.name = name
        self._check_valid()

    def _check_valid(self, raise_exception=True):
        if not isinstance(self.index, int):
            if raise_exception:
                raise IndexError('Wrong index name: has to be an int')
            else:
                return False
        if (not isinstance(self.name, str)) and (not isinstance(self.name, unicode)):
            if raise_exception:
                raise NameError('Wrong column name: has to be a string')
            else:
                return False
        return True

[docs]    def json(self):
        self._check_valid()

        if self.index >= 0:
            return {'type': 'Column', 'index': self.index}
        if len(self.name) > 0:
            return {'type': 'Column', 'name': self.name}
        raise ValueError('Either column index < 0 or column name is empty')


    def __eq__(self, other):
        if other is None or (not isinstance(other, self.__class__)):
            return False
        if (not self._check_valid(False)) or (not other._check_valid(False)):
            return False
        if (self.index >= 0) and (0 <= other.index == self.index):
            return True
        if len(self.name) > 0 and len(other.name) > 0 and self.name == other.name:
            return True
        return False

    def __str__(self):
        return 'ColumnOperand (index = %d, name = %s)' % (self.index, self.name)

    def __unicode__(self):
        return u'ColumnOperand (index = %d, name = %s)' % (self.index, self.name)

[docs]class SubsetExpression(object):
    """
    A subset expression for DDFs. The result of this expression is always a DDF.
    The result is lazily evaluated, and once evaluated, it will be cached.
    """
    # parameters
    ddf_id = None
    columns = []
    filter_condition = None

    _client = None
    _result = None

    def __init__(self, client, ddf_id, filter_condition=None, columns=None):
        if (client is None) or ddf_id is None:
            raise ValueError("Client and DDF ID must be specified")

        self._client = client
        self.ddf_id = ddf_id
        self.filter_condition = filter_condition
        self.columns = None if columns is None else columns[:]

    @property
    def result(self):
        """
        Get the result of this expression. The result is lazily evaluated
        :return: a DDF containing the result, or raise an exception if there is error
        """
        from dataframe import create_distributed_data_frame

        if self._result is None:
            filter_cond = None if self.filter_condition is None else self.filter_condition.json()
            res = self._client.exec_command('Subset',
                                            {'dataContainerID': self.ddf_id,
                                             'filter': filter_cond,
                                             'columns': [c.json() for c in self.columns]})['result']
            self._result = create_distributed_data_frame(self._client, res)
        return self._result

        return None if self.filter_condition is None else self.filter_condition.json()

    def _check_compatible_expression(self, other):
        """
        Check if the other object is compatible with this SubsetExpression
        To be compatible, they must have the same DDF ID and columns.
        :param other: the other SubsetExpression
        :return: NA.
        :exception: Raise various exceptions if the two expressions are not compatible
        """
        if other is None or (not isinstance(other, SubsetExpression)):
            raise ValueError('Incompatible subset expression, got type %s' % str(type(other)))

        if other.ddf_id is None or self.ddf_id != other.ddf_id:
            raise ValueError('Two SubsetExpression must operate on the same DDF')

        if len(self.columns) == len(other.columns) and all([(x in other.columns) for x in self.columns]):
            return
        else:
            raise ValueError('Both expressions must have either exactly the same set of columns')

    def __nonzero__(self):
        raise TypeError('Logical value of SubsetExpression is intentionally disabled.\n'
                        'Due to Python limitation,'
                        ' we can\'t override boolean operations "and", "or", "not", etc...\n'
                        'If you want to combine two SubsetExpressions using boolean operations, \n'
                        ' please use the bitwise operators "&", "|", "~" instead')

    def __bool__(self):
        """
        This is the python3 counterpart of __nonzero__
        :return: nothing
        """
        return self.__nonzero__()

    def __and__(self, other):

    def __invert__(self):
        if self.filter_condition is not None:
            return SubsetExpression(self._client, self.ddf_id,
                                    SubsetOperator(SubsetOperator.UNARY_OPERATOR_INVERT,
                                                   [self.filter_condition]),
                                    [c.copy() for c in self.columns])
        else:
            if len(self.columns) != 1:
                raise ValueError("We only support invert (~) on a single column")
            return SubsetExpression(self._client, self.ddf_id,
                                    SubsetOperator(SubsetOperator.UNARY_OPERATOR_INVERT,
                                                   [self.columns[0].copy()]),
                                    [])

[docs]    def contains(self, template, ignore_case=False):
        """
        :param template: a string to match with value in this subset
        :return: a SubsetExpression
        """
        util.check_type(template, "template", str, unicode)

        if self.filter_condition is not None or len(self.columns) != 1:
            raise ValueError('We only support "contains" on a single column')

        operator = SubsetOperator.BINARY_OPERATOR_GREP_IGNORE_CASE if \
            ignore_case else SubsetOperator.BINARY_OPERATOR_GREP
        return SubsetExpression(self._client, self.ddf_id,
                                SubsetOperator(operator,
                                               [SubsetExpression._get_scalar_operand(template),
                                                self.columns[0].copy()]),
                                [])


    @staticmethod
    def _get_scalar_operand(other):
        if isinstance(other, int) or isinstance(other, long):
            return ScalarOperand(ScalarOperand.OP_TYPE_INT, other)
        if isinstance(other, float):
            return ScalarOperand(ScalarOperand.OP_TYPE_DOUBLE, other)
        if isinstance(other, str) or isinstance(other, unicode):
            return ScalarOperand(ScalarOperand.OP_TYPE_STRING, other)
        if isinstance(other, bool):
            return ScalarOperand(ScalarOperand.OP_TYPE_BOOL, other)
        raise ValueError('Only scalar values are supported')

    def _handle_binary_operator_logic(self, other, operator):
        self._check_compatible_expression(other)
        if self.filter_condition is None and other.filter_condition is None:
            final_filter = None
        elif self.filter_condition is None and other.filter_condition is not None:
            if len(self.columns) == 1:
                # convert this into an operand
                final_filter = SubsetOperator(operator,
                                              [self.columns[0].copy(), other.filter_condition.copy()])
            else:
                raise ValueError('Invalid combination of logic expressions')
        elif self.filter_condition is not None and other.filter_condition is None:
            if len(other.columns) == 1:
                # convert this into an operand
                final_filter = SubsetOperator(operator,
                                              [self.filter_condition.copy(), other.columns[0].copy()])
            else:
                raise ValueError('Invalid combination of logic expressions')
        else:
            final_filter = SubsetOperator(operator,
                                          [self.filter_condition.copy(), other.filter_condition.copy()])
        return SubsetExpression(self._client, self.ddf_id,
                                final_filter,
                                [c.copy() for c in self.columns])

    def _handle_binary_operator_scalar(self, other, operator):
        scalar_op = SubsetExpression._get_scalar_operand(other)
        if self.filter_condition is None and len(self.columns) == 1:
            column_op = self.columns[0].copy()
            return SubsetExpression(self._client, self.ddf_id,
                                    SubsetOperator(operator, [column_op, scalar_op]), [])
        else:
            raise ValueError('For binary operator, we only support a column and a scalar value')
