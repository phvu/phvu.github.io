def validate_column_generic(col_names, column, get_name=True):
    """

[docs]    Validate a column name or index, return the column name
    :param col_names: list of column names
    :param column: column name or index
    :return: column index
    """
    if type(column) is int:
        if column < 0 or column >= len(col_names):
            raise ValueError('Column index out of range: %d' % column)
        return col_names[column] if get_name else column
    elif (isinstance(column, str) or isinstance(column, unicode)) and column in col_names:
        return column if get_name else col_names.index(column)
    raise ValueError('Column not found: ' + str(column))
