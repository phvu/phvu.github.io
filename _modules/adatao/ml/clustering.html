[docs]def kmeans(data, cols=None, num_clusters=2, max_iterations=10, distance='euclidean',
           runs=1, init_steps=5, epsilon=1e-5, seed=42):
    """
    Train a kmeans model on a DDF
    :param data: DDF
    :param cols:
    :param num_clusters:
    :param max_iterations:
    :param distance:
    :param runs:
    :param init_steps:
    :param epsilon:
    :param seed:
    :return:
    """

    colnames = data.colnames
    col_idx = range(len(colnames)) if cols is None else util.parse_columns(colnames, cols)

    if distance not in ['euclidean', 'cosine']:
        raise ValueError('distance: expect "euclidean" or "cosine"')

    res = data.client.exec_command('KmeansPlus',
                                   {'dataContainerID': data.oid,
                                    'xCols': col_idx,
                                    'K': num_clusters,
                                    'numIterations': max_iterations,
                                    'initialCentroids': None,
                                    'initializationMode': 'random',
                                    'doPersistResult': True,
                                    'distanceMeasure': distance,
                                    'runs': runs,
                                    'initializationSteps': init_steps,
                                    'epsilon': epsilon,
                                    'random_seed': seed})
    return ModelFactory.from_json(res['result'])




[docs]def silhouette(data, label_column, distance='euclidean', sample_fraction=1.0, seed=42):
    """
    Compute Silhouette index of the result of clustering
    :param data:
    :param label_column:
    :param distance:
    :param sample_fraction:
    :param seed:
    :return:
    """
    label_idx = util.parse_column(data.colnames, label_column)
    if distance not in ['euclidean', 'cosine']:
        raise ValueError('distance: expect "euclidean" or "cosine"')

    res = data.client.exec_command('SilhouetteIndex', {'dataContainerID': data.oid,
                                                       'labelColumn': label_idx,
                                                       'distanceMeasure': distance,
                                                       'sampleFraction': sample_fraction,
                                                       'seed': seed})['result']
    return create_distributed_data_frame(data.client, res)
