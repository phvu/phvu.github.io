    def __init__(self, **kwargs):
        self.output_dir = tempfile.gettempdir()
        if 'output_dir' in kwargs:
            self.output_dir = kwargs['output_dir']


[docs]    def render(self, data):
        file_name = datetime.datetime.now().strftime('%Y_%m_%d_%H_%M_%S')
        file_path = os.path.join(self.output_dir, file_name + '.html')
        i = 0
        while os.path.exists(file_path):
            file_path = os.path.join(self.output_dir, file_name + '_%d.html' % i)
            i += 1
        f = open(file_path, 'w')
        write_output(f, data)
        f.flush()
        f.close()
        return file_path
