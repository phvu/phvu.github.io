[docs]    def testUnique(self):
        df1 = self.mtcars.unique()
        self.assertIsInstance(df1, DistributedDataFrame)
        self.assertEqual(df1.cols, self.mtcars.cols)
        self.assertEqual(df1.rows, self.mtcars.rows)        # no duplicated rows
        self.assertNotEqual(df1.oid, self.mtcars.oid)

        df2 = self.mtcars.count_unique_values(['mpg', 'cyl', 5])
        self.assertIsInstance(df2, pd.DataFrame)
        self.assertAlmostEqual(df2.loc[df2['column'] == 'mpg', 'unique_count'].iloc[0], 25)
        self.assertAlmostEqual(df2.loc[df2['column'] == 'cyl', 'unique_count'].iloc[0], 3)
        self.assertAlmostEqual(df2.loc[df2['column'] == 'wt', 'unique_count'].iloc[0], 29)



[docs]    def testNA(self):
        self.mtcars.is_mutable = False
        df1 = self.mtcars.drop_na(margin='row', how='any')
        self.assertEqual(df1.rows, self.mtcars.rows)        # nothing to drop

        n_cols = self.mtcars.cols
        df = self.mtcars.fill_na(value=100)
        self.assertEqual(df.cols, n_cols)

        df = self.mtcars.fill_na(agg_func='mean')
        self.assertEqual(df.cols, n_cols)
        df = self.mtcars.fill_na(agg_func='median')
        self.assertEqual(df.cols, n_cols)

        # I don't understand this
        df = self.mtcars.fill_na(columns_to_values={'cyl': 100})
        self.assertEqual(df.cols, n_cols)

        df = self.mtcars.fill_na(value=100, columns=['mpg'])
        self.assertEqual(df.cols, 1)
        self.assertEqual(df.rows, self.mtcars.rows)

        # this is likely to fail, not sure why

[docs]    def testSummary(self):
        d = self.mtcars.summary()
        self.assertIsInstance(d.stats, dict)
        self.assertEqual(len(d.stats), self.mtcars.cols)
        self.assertIsInstance(repr(d), str)
        self.assertIsInstance(d.server_response, dict)

        # summary with a factor column
        df1 = self.mtcars.as_factor('cyl')
        d = df1.summary()
        self.assertIsInstance(d.stats, dict)
        self.assertEqual(len(d.stats), self.mtcars.cols)
        self.assertIsInstance(repr(d), str)

        d = self.mtcars.asummary()
        self.assertIsInstance(d, dict)

        d = self.mtcars.five_nums()
        self.assertIsInstance(d, pd.DataFrame)
        self.assertEqual(len(d), self.mtcars.cols)
        self.assertEqual(len(d.columns), 5)

        d1 = self.mtcars.quantile('mpg', [0, 0.25, 0.5, 1])
        self.assertIsInstance(d1, pd.DataFrame)
